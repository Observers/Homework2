public ActionResult Index()
{
	StudentContext db = new StudentContext();
	return View(db.Students.ToList());
}

To create:
--------------------
[HttpPost]
public ActionResult create(Student student)
{
	//Creating object of context
	//to connect to database table for saving and retrieving data
	StudentContext db = new StudentContext();

	//adds a new row with provided student info
	db.Students.Add(student);
	//Save changes
	db.SaveChanges();

	ViewBag.message = "Student Data Saved Successfully.";

	return View();
}

To edit:
---------------------
//Display
public ActionResult edit(int id)
{
	StudentContext db = new StudentContext();
	Student student = db.Students.SingleOrDefault(s=>s.sID==id);
	return View(student);
}

//Save
[HttpPost]
public ActionResult edit(Student student)
{
	StudentContext db = new StudentContext();
	Student student_update = db.Students.SingleOrDefault(s=>s.sID==student.sID);
	student_update.sName = student.sName;
	student_update.sAddress = student.sAddress;
	student_update.sPhone = student.sPhone;
	db.SaveChanges();

	ViewBag.message = "Student info updated successfully.";

	return View(student_update);
}
